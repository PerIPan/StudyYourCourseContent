Database Schema

chunks — the core RAG table
🔑 id UUID PK
→ document_id UUID FK → documents (CASCADE delete)
content TEXT ~500 tokens of source text
embedding VECTOR(1536) pgvector — OpenAI text-embedding-3-small
page_number INT page/slide where chunk starts
chunk_index INT order within document
created_at TIMESTAMPTZ
Indexes
CREATE INDEX ON chunks USING ivfflat (embedding vector_cosine_ops) WITH (lists = 10);
CREATE INDEX ON chunks (document_id);
CREATE INDEX ON documents (course_id);
CREATE UNIQUE INDEX ON courses (slug);

RAG Query (simplified)

-- 1. Embed user question via OpenAI
-- 2. Search chunks by cosine similarity

SELECT c.content, c.page_number,
  d.filename, d.lecture_number,
  co.name AS course_name,
  1 - (c.embedding <=> $1) AS similarity
FROM chunks c
JOIN documents d ON c.document_id = d.id
JOIN courses co ON d.course_id = co.id
WHERE 1 - (c.embedding <=> $1) > 0.7
  -- optional: AND co.slug = $2 (course filter)
ORDER BY c.embedding <=> $1
LIMIT 5;
